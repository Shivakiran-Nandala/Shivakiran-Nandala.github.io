import torch.nn as nn
import torch.nn.functional as F
from torch_geometric.nn import GCNConv

class GCNModel(nn.Module):
    def __init__(self, input_dim, hidden_dim, output_dim):
        super(GCNModel, self).__init__()
        self.conv1 = GCNConv(input_dim, hidden_dim)
        self.conv2 = GCNConv(hidden_dim, output_dim)

    def forward(self, data):
        x, edge_index = data.x, data.edge_index
        
        x = self.conv1(x, edge_index)
        x = F.relu(x)

        x = self.conv2(x, edge_index)
        
        return F.log_softmax(x, dim=1)

input_dim = X_tensor.shape[1]

from sklearn.metrics import make_scorer

def pr_auc_scorer(y_true, y_score):
    try:
        return average_precision_score(y_true, y_score)
    except Exception:
        return 0.0

def proba_scorer(estimator, X):
    if hasattr(estimator, "predict_proba"):
        return estimator.predict_proba(X)[:,1]
    elif hasattr(estimator, "decision_function"):
        return estimator.decision_function(X)
    else:
        return estimator.predict(X).astype(float)

pr_auc = make_scorer(average_precision_score, response_method="predict_proba")

COST_FN = 10.0  
COST_FP = 1.0  

precision, recall, thresh = precision_recall_curve(y_val, val_scores)
def eval_threshold(th):
    pred = (val_scores >= th).astype(int)
    cm = confusion_matrix(y_val, pred)
    tn, fp, fn, tp = cm.ravel()
    cost = COST_FN*fn + COST_FP*fp
    prec = precision_recall_fscore_support(y_val, pred, average="binary", zero_division=0)
    return {"threshold": th, "cost": cost, "precision": prec[0], "recall": prec[1], "f1": prec[2],
            "tp": int(tp), "fp": int(fp), "fn": int(fn), "tn": int(tn)}

from IPython.display import display, HTML

scores = test_scores

out = widgets.Output()

def render(th):
    pred = (scores >= th).astype(int)
    tn, fp, fn, tp = confusion_matrix(y_true, pred).ravel()
    prec, rec, f1, _ = precision_recall_fscore_support(y_true, pred, average="binary", zero_division=0)
    html = f"""
    <h4>Metrics at threshold = {th:.3f}</h4>
    <ul>
      <li>Precision: {prec:.4f}</li>
      <li>Recall: {rec:.4f}</li>
      <li>F1: {f1:.4f}</li>
      <li>TP/FP/FN/TN: {tp}/{fp}/{fn}/{tn}</li>
    </ul>
    """
    with out:
        out.clear_output()
        display(HTML(html))

def on_change(change):
    if change["name"] == "value":
        render(change["new"])

slider.observe(on_change)
