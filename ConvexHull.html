Project 1: Counting Sort Implementation

This project demonstrates the implementation of the **Counting Sort algorithm** in Java. Counting Sort is an efficient algorithm for sorting integers when the range of numbers is not significantly larger than the number of elements.

Features:

  • Reads input dynamically from the user
  • Finds the maximum value in the array
  • Uses counting technique for efficient sorting
  • Outputs the sorted array

Source Code:

// Importing Libraries to read and Process Code...
import java.util.ArrayList;
import java.util.Comparator;
import java.util.List;
import java.util.Random;
import java.util.Scanner;
import java.util.Stack;

// Defining the Points in 2D...
class Point2D {
    int x;
    int y;

    public Point2D(int x, int y) {
        this.x = x;
        this.y = y;
    }
}

// Implimenting Convex Hull in main ...
public class ConvexHull {
    public static void main(String[] args) {
        try (Scanner scanner = new Scanner(System.in)) {
            // Getting Range from user...
            System.out.print("Enter the range (minimum and maximum values) [in this Format: 0 100]: ");
            int Minrange = scanner.nextInt();
            int Maxrange = scanner.nextInt();

            // Number of Points to generate...
            System.out.print("Enter how many points want to generate: ");
            int size = scanner.nextInt();

            // Displaying the randomly Generated points...
            List points = RandomPoints(size, Minrange, Maxrange);
            System.out.println("\nGenerated points:");
            for (Point2D point : points) {
                System.out.println("(" + point.x + ", " + point.y + ")");
            }

            // Displaying the Convex Hull points...
            List convexHullPoints = GrahamScan(points);
            System.out.println("\nConvex Hull Points:");
            for (Point2D point : convexHullPoints) {
                System.out.println("(" + point.x + ", " + point.y + ")");
            }
        }
    }

    // Convex Hulll to generate Random points within the Range...
    public static List RandomPoints(int size, int Minrange, int Maxrange) {
        List points = new ArrayList<>();
        Random random = new Random();
        for (int m = 0; m < size; m++) {
            int x = random.nextInt(Maxrange - Minrange + 1) + Minrange;
            int y = random.nextInt(Maxrange - Minrange + 1) + Minrange;
            points.add(new Point2D(x, y));
        }
        return points;
    }

    // Graham Scan method for finding the convex hull points from random points...
    public static List GrahamScan(List points) {
        Point2D p0 = findLeftmostLowestPoint(points);
        List sortedPoints = sortPointsByPolarAngle(points, p0);
        Stack stack = new Stack<>();

        stack.push(p0);
        stack.push(sortedPoints.get(0));
        stack.push(sortedPoints.get(1));

        for (int i = 2; i < sortedPoints.size(); i++) {
            while (stack.size() >= 2
                    && !turnsLeft(stack.elementAt(stack.size() - 2), stack.peek(), sortedPoints.get(i))) {
                stack.pop();
            }
            stack.push(sortedPoints.get(i));
        }

        return new ArrayList<>(stack);
    }

    // implimenting leftmostlowestpoint method to find the left most point from the input points...
    public static Point2D findLeftmostLowestPoint(List points) {
        Point2D leftmostLowest = points.get(0);
        for (int i = 1; i < points.size(); i++) {
            Point2D currentPoint = points.get(i);
            if (currentPoint.y < leftmostLowest.y
                    || (currentPoint.y == leftmostLowest.y && currentPoint.x < leftmostLowest.x)) {
                leftmostLowest = currentPoint;
            }
        }
        return leftmostLowest;
    }

    // implimenting sortPointsByPolarAngle method to sort the input points based on the polar angle from the left most point...
    public static List sortPointsByPolarAngle(List points, Point2D p0) {
        List sortedPoints = new ArrayList<>(points);
        // to compare the polar angles...
        sortedPoints.sort(new PolarAngleComparator(p0));
        return sortedPoints;
    }

    // determining to make a left turn or not from given three points...
    public static boolean turnsLeft(Point2D p1, Point2D p2, Point2D p3) {
        return (p2.x - p1.x) * (p3.y - p1.y) - (p2.y - p1.y) * (p3.x - p1.x) > 0;
    }

    // To implement PolarAngleComparator on the Polar angles of the points w.r.t left most point...

    static class PolarAngleComparator implements Comparator {
        private Point2D p0;

        public PolarAngleComparator(Point2D p0) {
            this.p0 = p0;
        }

        @Override
        public int compare(Point2D p1, Point2D p2) {
            int angleComparison = Double.compare(polarAngle(p0, p1), polarAngle(p0, p2));
            if (angleComparison == 0) {
                // If two points have the same polar angle, the one closer to p0 comes first
                double dist1 = distance(p0, p1);
                double dist2 = distance(p0, p2);
                return Double.compare(dist1, dist2);
            }
            return angleComparison;
        }

        private double polarAngle(Point2D p0, Point2D p) {
            double dx = p.x - p0.x;
            double dy = p.y - p0.y;
            return Math.atan2(dy, dx);
        }

        private double distance(Point2D p1, Point2D p2) {
            return Math.sqrt((p1.x - p2.x) * (p1.x - p2.x) + (p1.y - p2.y) * (p1.y - p2.y));
        }
    }
}
