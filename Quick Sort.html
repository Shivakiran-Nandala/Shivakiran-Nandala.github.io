Project 1: Counting Sort Implementation

This project demonstrates the implementation of the **Counting Sort algorithm** in Java. Counting Sort is an efficient algorithm for sorting integers when the range of numbers is not significantly larger than the number of elements.

Features:

  • Reads input dynamically from the user
  • Finds the maximum value in the array
  • Uses counting technique for efficient sorting
  • Outputs the sorted array

Source Code:

import java.util.Arrays;
import java.util.Scanner;

//Main Function 
public class QUICK_SORT_20230502 {

    public static void main(String[] args) {

        Scanner input = new Scanner(System.in);

        // Taking Inputs [Number of Elements
        System.out.print("Enter the number the elements: ");

        // Creating InputArray of Length a...
        int n = input.nextInt();

        // Defining array of A is equal to the count of array
        int[] A = new int[n];
        // Taking Inputs Values from the User...
        System.out.println("Enter the number of elements:");

        // Storing Values in Array...
        for (int k = 0; k < n; k++) {
            A[k] = input.nextInt();
        }

        // Printing the input Array...
        System.out.println("Input array: " + Arrays.toString(A));

        quicksort(A, 0, A.length - 1);

        // Printing the Sorted Array...
        System.out.println("Sorted array: " + Arrays.toString(A));

        input.close();
    }

    public static void quicksort(int[] A, int p, int r) {
        if (p < r) {
            // Partitioning the array
            int q = partition(A, p, r);

            // Recursively sort after finding the partitioning element
            quicksort(A, p, q - 1);
            quicksort(A, q + 1, r);
        }
    }

    public static int partition(int[] A, int p, int r) {
        // defining x variariable as pivot
        int x = A[r];

        int i = p - 1;

        for (int j = p; j < r; j++) {
            // Creating an if condition to check the current element is smaller or equal to
            // pivot or not.
            if (A[j] <= x) {
                i = i + 1;

                int temporary = A[i];
                A[i] = A[j];
                A[j] = temporary;
            }
        }

        int temporary = A[i + 1];
        A[i + 1] = A[r];
        A[r] = temporary;

        return i + 1;
    }
}
