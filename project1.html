Project 1: Counting Sort Implementation

This project demonstrates the implementation of the **Counting Sort algorithm** in Java. Counting Sort is an efficient algorithm for sorting integers when the range of numbers is not significantly larger than the number of elements.

Features:

  • Reads input dynamically from the user
  • Finds the maximum value in the array
  • Uses counting technique for efficient sorting
  • Outputs the sorted array

Source Code:

// Importing Libraries to read and Process Code...
import java.util.Arrays;
import java.util.Scanner;

public class CountingSortCode {
    public static void main(String[] args) {
        Scanner input = new Scanner(System.in);
        System.out.print("Enter the number of elements: ");
        int a = input.nextInt();
        int[] inputarray = new int[a];

        System.out.println("Enter the elements:");
        for (int i = 0; i < a; i++) {
            inputarray[i] = input.nextInt();
        }
        input.close();

        int k = inputarray[0];
        for (int i = 0; i < inputarray.length; i++) {
            if (inputarray[i] > k)
                k = inputarray[i];
        }
        System.out.println("Maximum value in the Array: " + k);

        int[] sortedarray = new int[a];
        int[] proarray = new int[k + 1];

        for (int i = 0; i < a; i++) {
            proarray[inputarray[i]]++;
        }
        System.out.println("Proarray after counting occurances: " + Arrays.toString(proarray));

        for (int i = 1; i < k + 1; i++) {
            proarray[i] += proarray[i - 1];
        }
        System.out.println("Processing array: " + Arrays.toString(proarray));

        for (int i = a - 1; i >= 0; i--) {
            sortedarray[proarray[inputarray[i]] - 1] = inputarray[i];
            proarray[inputarray[i]]--;
        }
        System.out.println("Sorted array: " + Arrays.toString(sortedarray));
    }
}
